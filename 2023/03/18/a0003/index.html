# SSH設定

Ubuntu22.04が初期状態ではTeratermからログインできない問題の対応方法 (opens new window)

当該サーバへSSHログイン

ssh ubuntu@[Indigo Server IP] -i /home/ubuntu/[SSH_key]
# Update Ubuntu
sudo apt update
sudo apt upgrade
# Modify sshd_config
sudo cp -p /etc/ssh/sshd_config /etc/ssh/sshd_config.bkup
sed -i '1s/^/PubkeyAcceptedAlgorithms=+ssh-rsa\n/' /etc/ssh/sshd_config
sudo vi /etc/ssh/sshd_config

/etc/ssh/sshd_config

- Port 22
+ Port [任意のポート]

sudo reboot

以降は、当該サーバIP:[任意のポート]へSSH接続

# mastodon導入

# docker インストール

sudo su -
snap install docker

# Git clone

git clone https://github.com/tootsuite/mastodon.git
cd mastodon/
git tag -l
git checkout v4.0.2 # その時の最新の安定版を選択

# docker-compose.ymlの修正

vi docker-compose.yml

docker-compose.yml

    environment:
      - 'POSTGRES_HOST_AUTH_METHOD=trust'
+      - 'POSTGRES_USER=mastodon'
+      - 'POSTGRES_DB=mastodon_production'
+      - 'POSTGRES_PASSWORD=[任意のパスワード]'

web:
+  #build: .
streaming:
+  #build: .
sidekiq:
+  #build: .

# mastodon設定の修正

cp .env.production.sample .env.production
vi .env.production

.env.production

-LOCAL_DOMAIN=example.com
+LOCAL_DOMAIN=[ドメイン名]

-REDIS_HOST=localhost
+REDIS_HOST=redis

-DB_HOST=/var/run/postgresql
+DB_HOST=db

-DB_PASS=
+DB_PASS=[DBパスワード]

-ES_PASS=password
+ES_PASS=[ESパスワード]

-SMTP_SERVER=smtp.mailgun.org
+SMTP_SERVER=smtp.gmail.com

-SMTP_LOGIN=
-SMTP_PASSWORD=
-SMTP_FROM_ADDRESS=notifications@example.com
+SMTP_LOGIN=[Gmainアドレス]
+SMTP_PASSWORD=[Gmailアプリパスワード]
+SMTP_FROM_ADDRESS=[Gmainアドレス]

-S3_ENABLED=true
+S3_ENABLED=false

# docker image のPull

docker-compose pull

# Secrets作成

表示されたキーの1回目をSECRET_KEY_BASE、2回目をOTP_SECRETとして追加

docker-compose run --rm web bundle exec rake secret
docker-compose run --rm web bundle exec rake secret
vi .env.production

.env.production

+ SECRET_KEY_BASE=[1回目のコマンド結果]
+ OTP_SECRET=[2回目のコマンド結果]

# Web Push用の鍵を作成

docker compose run --rm web bundle exec rake mastodon:webpush:generate_vapid_key

+ VAPID_PRIVATE_KEY=[コマンド実行結果]
+ VAPID_PUBLIC_KEY=[コマンド実行結果]

# DB作成

docker compose exec db bash
bash-5.1# psql -U postgres

postgres=# CREATE USER mastodon;
CREATE ROLE
postgres=# alter role mastodon_user with password '[DBパスワード]';
ALTER ROLE
postgres=# create database mastodon_production with owner=mastodon;
CREATE DATABASE
postgres=# \q

docker-compose run --rm web rails db:migrate

# アセットファイル作成

docker-compose run --rm web rails assets:precompile

# dockerコンテナ（Mastodon）立ち上げ

docker-compose up -d

# Nginxリバースプロキシ設定

# nginx、certbot導入

apt -y install nginx
apt -y install certbot

# Let's Encrypt設定

プロンプトに従い設定を行う ※自動更新については別途実装が必要

certbot certonly --webroot -w /var/www/html -d [ドメイン名]

# リバースプロキシ設定

cp dist/nginx.conf /etc/nginx/sites-available/mastodon
vi /etc/nginx/sites-available/mastodon

/etc/nginx/sites-available/mastodon

 server {
   listen 80;
   listen [::]:80;
-  server_name example.com;
+  server_name [ドメイン名];

 server {
   listen 443 ssl http2;
   listen [::]:443 ssl http2;
-  server_name example.com;
+  server_name [ドメイン名];

-  # ssl_certificate     /etc/letsencrypt/live/example.com/fullchain.pem;
-  # ssl_certificate_key /etc/letsencrypt/live/example.com/privkey.pem;
+   ssl_certificate     /etc/letsencrypt/live/[ドメイン名]/fullchain.pem;
+   ssl_certificate_key /etc/letsencrypt/live/[ドメイン名]/privkey.pem;

   location = /sw.js {
-    try_files $uri =404;

# アカウント作成

ブラウザからアカウント作成する
http://[ドメイン名]

作成したアカウントに管理者権限を与える

docker-compose exec web bin/tootctl accounts modify [ユーザ名] --role Admin

# 新規アカウントの作成を禁止する

[管理]>[サーバ設定]>[アカウント作成]>[新規登録が可能な人]
「誰にも許可しない」に変更して保存
